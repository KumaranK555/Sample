SELECT Newf2,Newf3,Newf4,sum(CompanyCurrency_HSL) as ActualFTM from GLPCP where tabflag='Actual' and Product !='JJ'
 and FiscalYear_Ryear = (select Max(FiscalYear_RYEAR) from GLPCP ) and NPeriod =8 and Months='Aug' 
 group by Newf2,NEWF3,NEWF4


select Newf2,Newf3,Newf4,sum(CompanyCurrency_HSL) as JobWorkFTM  from GLPCP 
where Origin_RHOART =03 and Product !='JJ' and FiscalYear_Ryear = (select Max(FiscalYear_RYEAR) from GLPCP )
 and NPeriod = 8  and Months='Aug' group by Newf2,NEWF3,NEWF4
